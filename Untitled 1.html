<script type="module">

};


// --- FIREBASE SETUP ---

const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-finance-app';

const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

const app = initializeApp(firebaseConfig);

state.db = getFirestore(app);

state.auth = getAuth(app);


// --- UI ELEMENTS ---

const ui = {

appContainer: document.getElementById('app-container'),

tabs: {

dashboard: document.getElementById('tab-dashboard'),

},

views: {

},

dashboard: {

currentMonthDisplay: document.getElementById('current-month-display'),

prevMonthBtn: document.getElementById('prev-month-btn'),

nextMonthBtn: document.getElementById('next-month-btn'),

kpiBalance: document.getElementById('kpi-balance'),

kpiTotalSpent: document.getElementById('kpi-total-spent'),

kpiYourShare: document.getElementById('kpi-your-share'),

kpiPersonalSpent: document.getElementById('kpi-personal-spent'),

kpiSharedSpent: document.getElementById('kpi-shared-spent'),

kpiRecurringSpent: document.getElementById('kpi-recurring-spent'),

chartCanvas: document.getElementById('expenses-chart'),

chartFilterButtons: document.getElementById('chart-filter-buttons'),

monthlyIncomeInput: document.getElementById('monthly-income'),

saveSettingsBtn: document.getElementById('save-settings-btn'),

recurringExpensesList: document.getElementById('recurring-expenses-list'),

addRecurringForm: document.getElementById('add-recurring-form'),

launchRecurringBtn: document.getElementById('launch-recurring-btn'),

categoriesList: document.getElementById('categories-list'),

addCategoryForm: document.getElementById('add-category-form'),

},

transactions: {

aiPromptInput: document.getElementById('ai-prompt-input'),

aiFillBtn: document.getElementById('ai-fill-btn'),

aiBtnText: document.getElementById('ai-btn-text'),

aiBtnSpinner: document.getElementById('ai-btn-spinner'),

form: document.getElementById('transaction-form'),

formId: document.getElementById('transaction-id'),

formDate: document.getElementById('transaction-date'),

formDescription: document.getElementById('transaction-description'),

formCategory: document.getElementById('transaction-category'),

},

}

};


// --- UTILS ---

const formatCurrency = (value) => new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' }).format(value);

const showToast = (message, type = 'success') => {

ui.toast.message.textContent = message;

ui.toast.element.classList.remove('opacity-0', 'translate-y-10', 'bg-green-100', 'bg-red-100', 'text-green-700', 'text-red-700');

if (type === 'success') {

ui.toast.icon.innerHTML = '✅';

ui.toast.element.classList.add('bg-green-100', 'text-green-700');

} else {

ui.toast.icon.innerHTML = '❌';

ui.toast.element.classList.add('bg-red-100', 'text-red-700');

}

ui.toast.element.classList.add('opacity-100', 'translate-y-0');


setTimeout(() => {

ui.toast.element.classList.add('opacity-0', 'translate-y-10');

}, 3000);

};

const getFirestorePaths = () => {

if (!state.userId) return null;

return {

settings: doc(state.db, `artifacts/${appId}/users/${state.userId}/settings/config`),

transactions: collection(state.db, `artifacts/${appId}/users/${state.userId}/transactions`)

};

};


// --- RENDERING FUNCTIONS ---


const render = () => {

if (!state.isAuthReady) return;

renderTabs();

if (state.currentTab === 'dashboard') {

renderDashboard();

} else {

renderTransactionsView();

}

};


const renderTabs = () => {

Object.values(ui.tabs).forEach(tab => tab.classList.remove('tab-active'));

Object.values(ui.views).forEach(view => view.classList.add('hidden'));


ui.tabs[state.currentTab].classList.add('tab-active');

ui.views[state.currentTab].classList.remove('hidden');

};


const renderDashboard = () => {

// Month Display

ui.dashboard.currentMonthDisplay.textContent = state.currentDate.format('MMMM [de] YYYY');


// Filter transactions for the current month

const startOfMonth = state.currentDate.startOf('month');

const endOfMonth = state.currentDate.endOf('month');

const monthTransactions = state.transactions.filter(t => {

const tDate = dayjs(t.date);

return tDate.isAfter(startOfMonth.subtract(1, 'day')) && tDate.isBefore(endOfMonth.add(1, 'day'));

});


// Calculate KPIs

const totalSpent = monthTransactions.reduce((sum, t) => sum + t.value, 0);

const personalSpent = monthTransactions.filter(t => t.type === 'personal').reduce((sum, t) => sum + t.value, 0);

const sharedSpent = monthTransactions.filter(t => t.type === 'shared').reduce((sum, t) => sum + t.value, 0);

const recurringSpent = monthTransactions.filter(t => t.isRecurring).reduce((sum, t) => sum + t.value, 0);

const yourShare = personalSpent + (sharedSpent / 2);

const balance = state.settings.monthlyIncome - totalSpent;


// Update KPI elements

ui.dashboard.kpiBalance.textContent = formatCurrency(balance);

ui.dashboard.kpiBalance.className = `text-2xl font-semibold mt-1 ${balance >= 0 ? 'text-green-600' : 'text-red-600'}`;

ui.dashboard.kpiTotalSpent.textContent = formatCurrency(totalSpent);

ui.dashboard.kpiYourShare.textContent = formatCurrency(yourShare);

ui.dashboard.kpiPersonalSpent.textContent = formatCurrency(personalSpent);

ui.dashboard.kpiSharedSpent.textContent = formatCurrency(sharedSpent);

ui.dashboard.kpiRecurringSpent.textContent = formatCurrency(recurringSpent);

// Check if recurring expenses for this month have been launched

const firstDayOfMonthStr = state.currentDate.startOf('month').format('YYYY-MM-DD');

const hasLaunched = state.transactions.some(t => t.isRecurring && t.date.startsWith(firstDayOfMonthStr));

ui.dashboard.launchRecurringBtn.disabled = hasLaunched;

ui.dashboard.launchRecurringBtn.textContent = hasLaunched ? 'Recorrentes já lançados' : 'Lançar Recorrentes do Mês';


renderChart(monthTransactions);

renderSettings();

};


const renderChart = (transactions) => {

const filteredTransactions = transactions.filter(t => {

if (state.chartFilter === 'all') return true;

return t.type === state.chartFilter;

});


const dataByCat = filteredTransactions.reduce((acc, t) => {

acc[t.category] = (acc[t.category] || 0) + t.value;

return acc;

}, {});


const labels = Object.keys(dataByCat);

const data = Object.values(dataByCat);


const chartData = {

labels: labels,

datasets: [{

label: 'Gastos por Categoria',

data: data,

backgroundColor: [

'#6366f1', '#818cf8', '#a5b4fc', '#c7d2fe',

'#f43f5e', '#fb7185', '#fda4af', '#fecdd3',

'#10b981', '#34d399', '#6ee7b7', '#a7f3d0'

],

hoverOffset: 4

}]

};


if (state.chartInstance) {

state.chartInstance.data = chartData;

state.chartInstance.update();

} else {

state.chartInstance = new Chart(ui.dashboard.chartCanvas, {

type: 'doughnut',

data: chartData,

options: {

responsive: true,

maintainAspectRatio: false,

plugins: {

legend: {

position: 'bottom',

},

tooltip: {

callbacks: {

label: function(context) {

let label = context.label || '';

if (label) {

label += ': ';

}

if (context.parsed !== null) {

label += formatCurrency(context.parsed);

}

return label;

}

}

}

}

}

});

}

};

const renderSettings = () => {

// Income

ui.dashboard.monthlyIncomeInput.value = state.settings.monthlyIncome || '';


// Categories

ui.dashboard.categoriesList.innerHTML = '';

state.settings.categories.forEach(cat => {

const el = document.createElement('div');

el.className = 'flex justify-between items-center bg-slate-50 p-2 rounded';

el.innerHTML = `

<span class="text-sm">${cat}</span>

<button data-category="${cat}" class="delete-category-btn text-slate-400 hover:text-red-500 transition-colors">

<span class="material-symbols-outlined text-base">delete</span>

const renderTransactionsView = () => {

// Update view switcher buttons

ui.transactions.viewSwitcherBtns.forEach(btn => {

btn.classList.toggle('bg-white', btn.dataset.view === state.transactionsView);

btn.classList.toggle('text-indigo-600', btn.dataset.view === state.transactionsView);

btn.classList.toggle('shadow-sm', btn.dataset.view === state.transactionsView);

btn.classList.toggle('text-slate-600', btn.dataset.view !== state.transactionsView);

});


// Update period display and filter transactions

let transactionsToDisplay = [];

if (state.transactionsView === 'daily') {

ui.transactions.currentPeriodDisplay.textContent = state.currentDate.format('DD [de] MMMM, YYYY');

transactionsToDisplay = state.transactions.filter(t => dayjs(t.date).isSame(state.currentDate, 'day'));

} else { // weekly

const startOfWeek = state.currentDate.startOf('week');

const endOfWeek = state.currentDate.endOf('week');

ui.transactions.currentPeriodDisplay.textContent = `${startOfWeek.format('DD/MM')} - ${endOfWeek.format('DD/MM/YYYY')}`;

transactionsToDisplay = state.transactions.filter(t => {

const tDate = dayjs(t.date);

return tDate.isAfter(startOfWeek.subtract(1, 'day')) && tDate.isBefore(endOfWeek.add(1, 'day'));

});

}

// Sort transactions by date (most recent first)

transactionsToDisplay.sort((a, b) => dayjs(b.date).valueOf() - dayjs(a.date).valueOf());


// Render transaction cards

ui.transactions.container.innerHTML = '';

if (transactionsToDisplay.length === 0) {

ui.transactions.container.innerHTML = `<p class="text-center text-slate-500 py-10">Nenhum lançamento para este período.</p>`;

return;

}


transactionsToDisplay.forEach(t => {

const card = document.createElement('div');

card.className = 'flex items-center justify-between p-4 border-b border-slate-100 hover:bg-slate-50';

const typeIcon = t.type === 'personal'

? `<span class="material-symbols-outlined text-blue-500" title="Pessoal">person</span>`

: `<span class="material-symbols-outlined text-purple-500" title="Compartilhado">groups</span>`;

card.innerHTML = `

<div class="flex items-center gap-4">

${typeIcon}

<div>

<p class="font-semibold text-slate-800">${t.description}</p>

<p class="text-sm text-slate-500">${t.category} &bull; ${dayjs(t.date).format('DD/MM/YY')}</p>

</div>

</div>

<div class="flex items-center gap-4">

<p class="font-semibold text-red-500">${formatCurrency(t.value)}</p>

<div class="flex gap-1">

<button data-id="${t.id}" class="edit-transaction-btn p-2 rounded-full hover:bg-slate-200 transition-colors">

<span class="material-symbols-outlined text-lg text-slate-600">edit</span>

</button>

<button data-id="${t.id}" class="delete-transaction-btn p-2 rounded-full hover:bg-slate-200 transition-colors">

<span class="material-symbols-outlined text-lg text-slate-600">delete</span>

</button>

</div>

</div>

`;

ui.transactions.container.appendChild(card);

});

};


// --- DATA HANDLING (FIRESTORE) ---

const setupFirestoreListeners = () => {

const paths = getFirestorePaths();

if (!paths) return;


// Settings listener

onSnapshot(paths.settings, (docSnap) => {

if (docSnap.exists()) {

state.settings = { ...state.settings, ...docSnap.data() };

} else {

// Create initial settings doc if it doesn't exist

setDoc(paths.settings, state.settings);

}

render();

}, (error) => console.error("Error listening to settings:", error));


// Transactions listener

onSnapshot(paths.transactions, (snapshot) => {

state.transactions = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));

render();

}, (error) => console.error("Error listening to transactions:", error));

};


const saveSettings = async () => {

const paths = getFirestorePaths();

if (!paths) return;

try {

const income = parseFloat(ui.dashboard.monthlyIncomeInput.value) || 0;

await updateDoc(paths.settings, { monthlyIncome: income });

showToast('Renda salva com sucesso!');

} catch (error) {

console.error("Error saving income:", error);

showToast('Erro ao salvar renda.', 'error');

}

};


const addCategory = async (name) => {

const paths = getFirestorePaths();

if (!paths || !name) return;

try {

await updateDoc(paths.settings, { categories: arrayUnion(name) });

showToast('Categoria adicionada!');

} catch (error) {

console.error("Error adding category:", error);

showToast('Erro ao adicionar categoria.', 'error');

}

};


const deleteCategory = async (name) => {

const paths = getFirestorePaths();

if (!paths || !name) return;

try {

await updateDoc(paths.settings, { categories: arrayRemove(name) });

showToast('Categoria removida!');

} catch (error) {

console.error("Error deleting category:", error);

showToast('Erro ao remover categoria.', 'error');

}

};

const addRecurringExpense = async (expense) => {

const paths = getFirestorePaths();

if (!paths || !expense) return;

try {

await updateDoc(paths.settings, { recurringExpenses: arrayUnion(expense) });

showToast('Gasto recorrente adicionado!');

} catch (error) {

console.error("Error adding recurring expense:", error);

showToast('Erro ao adicionar gasto recorrente.', 'error');

}

};

const deleteRecurringExpense = async (expenseName) => {

const paths = getFirestorePaths();

if (!paths || !expenseName) return;

const expenseToDelete = state.settings.recurringExpenses.find(e => e.name === expenseName);

if (!expenseToDelete) return;

try {

await updateDoc(paths.settings, { recurringExpenses: arrayRemove(expenseToDelete) });

showToast('Gasto recorrente removido!');

} catch (error) {

console.error("Error deleting recurring expense:", error);

showToast('Erro ao remover gasto recorrente.', 'error');

}

};

const launchRecurringExpenses = async () => {

const paths = getFirestorePaths();

if (!paths) return;

const firstDayOfMonth = state.currentDate.startOf('month').format('YYYY-MM-DD');

try {

for (const expense of state.settings.recurringExpenses) {

const newTransaction = {

date: firstDayOfMonth,

description: expense.name,

category: expense.category || 'Recorrente',

value: parseFloat(expense.value),

type: expense.type,

isRecurring: true,

createdAt: new Date().toISOString()

};

await addDoc(paths.transactions, newTransaction);

}

showToast('Gastos recorrentes lançados com sucesso!');

} catch (error) {

console.error("Error launching recurring expenses:", error);

showToast('Erro ao lançar gastos recorrentes.', 'error');

}

};


const saveTransaction = async (transactionData) => {

const paths = getFirestorePaths();

if (!paths) return;


const id = ui.transactions.formId.value;

try {

if (id) { // Update

const transactionRef = doc(state.db, `artifacts/${appId}/users/${state.userId}/transactions`, id);

await setDoc(transactionRef, transactionData, { merge: true });

showToast('Lançamento atualizado com sucesso!');

} else { // Create

await addDoc(paths.transactions, { ...transactionData, createdAt: new Date().toISOString() });

showToast('Lançamento adicionado com sucesso!');

}

resetTransactionForm();

} catch (error) {

console.error("Error saving transaction:", error);

showToast('Erro ao salvar lançamento.', 'error');

}

};


const deleteTransaction = async (id) => {

if (!confirm('Tem certeza que deseja excluir este lançamento?')) return;

const paths = getFirestorePaths();

if (!paths) return;

try {

const transactionRef = doc(state.db, `artifacts/${appId}/users/${state.userId}/transactions`, id);

await deleteDoc(transactionRef);

showToast('Lançamento excluído.');

} catch (error) {

console.error("Error deleting transaction:", error);

showToast('Erro ao excluir lançamento.', 'error');

}

};

// --- GEMINI AI ---

const handleAIFill = async () => {

const promptText = ui.transactions.aiPromptInput.value;

if (!promptText.trim()) {

showToast('Por favor, descreva o gasto primeiro.', 'error');

return;

}


ui.transactions.aiBtnText.classList.add('hidden');

ui.transactions.aiBtnSpinner.classList.remove('hidden');

ui.transactions.aiFillBtn.disabled = true;


const prompt = `

Analise o seguinte texto sobre um gasto e extraia a descrição, o valor numérico e a categoria.

Texto do usuário: "${promptText}"

Categorias disponíveis: ${state.settings.categories.join(', ')}.

Se não encontrar uma categoria exata, escolha a mais apropriada da lista.

Responda APENAS com um objeto JSON.

`;

const payload = {

contents: [{ role: "user", parts: [{ text: prompt }] }],

generationConfig: {

responseMimeType: "application/json",

responseSchema: {

type: "OBJECT",

properties: {

description: { type: "STRING" },

value: { type: "NUMBER" },

category: { type: "STRING" }

},

required: ["description", "value", "category"]

}

}

};


const apiKey = ""; // Provided by the environment

const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;


try {

const response = await fetch(apiUrl, {

method: 'POST',

headers: { 'Content-Type': 'application/json' },

body: JSON.stringify(payload)

});


if (!response.ok) {

throw new Error(`API Error: ${response.statusText}`);

}


const result = await response.json();

const text = result?.candidates?.[0]?.content?.parts?.[0]?.text;


if (text) {

const data = JSON.parse(text);

ui.transactions.formDescription.value = data.description || '';

ui.transactions.formValue.value = data.value || '';

ui.transactions.formCategory.value = data.category || state.settings.categories[0];

showToast('Campos preenchidos pela IA!');

} else {

throw new Error('Resposta da IA inválida.');

}


} catch (error) {

console.error("AI Fill Error:", error);

showToast('Não foi possível processar a solicitação.', 'error');

} finally {

ui.transactions.aiBtnText.classList.remove('hidden');

ui.transactions.aiBtnSpinner.classList.add('hidden');

ui.transactions.aiFillBtn.disabled = false;

}

};

const init = async () => {

// Setup Day.js

dayjs.extend(dayjs_plugin_utc);

dayjs.extend(dayjs_plugin_weekday);

dayjs.extend(dayjs_plugin_localeData);

dayjs.locale('pt-br');


// Set initial form date

ui.transactions.formDate.value = dayjs().format('YYYY-MM-DD');

// Authenticate user and setup listeners

onAuthStateChanged(state.auth, (user) => {

if (user) {

state.userId = user.uid;

state.isAuthReady = true;

setupFirestoreListeners();

ui.appContainer.classList.remove('opacity-0');

}

});


try {

await setPersistence(state.auth, browserLocalPersistence);

if (initialAuthToken) {

await signInWithCustomToken(state.auth, initialAuthToken);

} else if (!state.auth.currentUser) {

await signInAnonymously(state.auth);

}

} catch (error) {

console.error("Authentication Error:", error);

// Fallback if persistence fails or in restrictive environments

if (error.code === 'auth/persistence-unavailable') {

if (initialAuthToken) {

await signInWithCustomToken(state.auth, initialAuthToken);

} else if (!state.auth.currentUser) {

await signInAnonymously(state.auth);

}

}

}


addEventListeners();

render(); // Initial render

};

document.addEventListener('DOMContentLoaded', init);
